part 3

[y,Fs] = audioread('HW1_Q4_voice.wav');
n=length(y);
y=y(Fs:2*Fs);
y=y';
fftsignal=fft(y);
fftsignal = fftshift(fftsignal);

f=Fs/2*linspace(-pi,pi,length(fftsignal));
figure(3)
plot(f,abs(fftsignal))
title('Magnitude')
figure(4)
plot(angle(fftsignal))
title('Phase')
